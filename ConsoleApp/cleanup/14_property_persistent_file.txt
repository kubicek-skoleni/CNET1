
namespace ConsoleApp.Model
{
    public class Car
    {
        public string PlateNumber { get; set; }
        public int YearOfProduction { get; set; }
        public string Brand { get; set; }
        public string Model { get; set; }
        public string Color { get; set; }

        public double Mileage 
        {
            get
            {
                if(!File.Exists(PlateNumber + ".txt"))
                {
                    File.WriteAllText(PlateNumber + ".txt", "0");
                }

                string kms_string = File.ReadAllText(PlateNumber + ".txt");
                double kms = double.Parse(kms_string);
                return kms;
            }
        }

        public void AddKm(double km)
        {
            var new_mileage = Mileage + km;
            File.WriteAllText(PlateNumber + ".txt", new_mileage.ToString());
        }

        public void PrintInfo()
        {
            Console.WriteLine();
            Console.Write(Brand + " - ");
            Console.Write(Model + " - ");
            Console.Write(Color + " - ");
            Console.Write(PlateNumber);
        }
    }
}
